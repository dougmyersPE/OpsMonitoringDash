    Full implementation:
    1. Get sync Redis client: `redis_client = get_sync_redis()`
    2. Acquire distributed lock — use 120s timeout (not 60s — see RESEARCH.md Pitfall 3):
       ```python
       lock = redis_client.lock(
           f"lock:update_event_status:{event_id}",
           timeout=120,      # 120s: covers 3 retries × ProphetX API latency + safety margin
           blocking=False,   # non-blocking: if not acquired, log and return immediately
       )
       acquired = lock.acquire()
       if not acquired:
           log.info("update_event_status_lock_not_acquired", event_id=event_id, actor=actor)
           return  # Another worker is handling this — exit cleanly (not an error)
       ```
    3. Inside `try/finally` (lock released in finally):
       a. Open SyncSessionLocal
       b. Look up Event by `id` column (UUID) — if not found, log warning and return
       c. If `target_status` is None: derive from Event.real_world_status via mismatch_detector.get_expected_px_status(). If still None (flag-only status), log and return without action.
       d. **Idempotency guard:** if `event.prophetx_status == target_status`, log "already_at_target" and return without API call
       e. Record `before_state = {"prophetx_status": event.prophetx_status}`
       f. Call ProphetX status update — **stub for now** (ProphetX write endpoint unconfirmed):
          ```python
          # TODO: Wire real ProphetX write endpoint when confirmed
          # Expected: PATCH /mm/update_sport_event_status or similar

          log.info("update_event_status_stub_would_call_prophetx",
                   prophetx_event_id=event.prophetx_event_id,
                   target_status=target_status)
          px_success = True  # Stub: assume success
          ```
       g. Update local DB: `event.prophetx_status = target_status`
       h. Write audit log entry (INSERT) in same session:
          ```python

              action_type="status_update",
              actor=actor,
              entity_type="event",
              entity_id=event.id,
              before_state=before_state,
              after_state={"prophetx_status": target_status},
              result="success",
          )
          session.add(audit_entry)
          ```
       i. `session.commit()` — both event update and audit log written atomically
       j. `log.info("update_event_status_complete", event_id=event_id, before=before_state, after=target_status, actor=actor)`
    4. On exception: log error, write audit log with result="failure" and error_message, then `raise self.retry(exc=exc, countdown=2 ** self.request.retries)`
    5. `finally: lock.release()` — wrap in try/except (lock may have expired)

    **backend/app/api/v1/events.py:**
    ```python
    router = APIRouter(prefix="/events", tags=["events"])

    @router.get("", response_model=EventListResponse,
                dependencies=[Depends(require_role("read_only", "operator", "admin"))])
    async def list_events(session: AsyncSession = Depends(get_async_session)):
        total_q = await session.execute(select(func.count()).select_from(Event))
        total = total_q.scalar_one()
        rows_q = await session.execute(select(Event).order_by(Event.scheduled_start.asc()))
        events = rows_q.scalars().all()
        return EventListResponse(total=total, events=events)

    @router.post("/{event_id}/sync-status",
                 dependencies=[Depends(require_role("operator", "admin"))])
    async def manual_sync_status(
        event_id: str,
        current_user: dict = Depends(get_current_user),
        session: AsyncSession = Depends(get_async_session),
    ):
        """Manually trigger status sync. Enqueues same worker as auto-sync."""
        # Verify event exists
        event = await session.get(Event, uuid.UUID(event_id))
        if not event:
            raise HTTPException(status_code=404, detail="Event not found")
        update_status_task.delay(
            event_id=event_id,
            target_status=None,        # derive from real_world_status in worker
            actor=current_user["sub"], # user email as actor
        )
        return {"queued": True, "event_id": event_id}
    ```

    **backend/app/api/v1/markets.py:**
    ```python
    router = APIRouter(prefix="/markets", tags=["markets"])

    @router.get("", response_model=MarketListResponse,
                dependencies=[Depends(require_role("read_only", "operator", "admin"))])
    async def list_markets(session: AsyncSession = Depends(get_async_session)):
        total_q = await session.execute(select(func.count()).select_from(Market))
        total = total_q.scalar_one()
        rows_q = await session.execute(select(Market).order_by(Market.name.asc()))
        markets = rows_q.scalars().all()
        return MarketListResponse(total=total, markets=markets)

    @router.patch("/{market_id}/config",
                  response_model=MarketResponse,
                  dependencies=[Depends(require_role("admin"))])
    async def update_market_config(
        market_id: str,
        body: MarketConfigUpdate,
        session: AsyncSession = Depends(get_async_session),
    ):
        """Update per-market liquidity threshold. Set to null to restore global default fallback."""
        market = await session.get(Market, uuid.UUID(market_id))
        if not market:
            raise HTTPException(status_code=404, detail="Market not found")
        market.min_liquidity_threshold = body.min_liquidity_threshold
        await session.commit()
        await session.refresh(market)
        return market
    ```

    **backend/app/api/v1/audit.py:**
    ```python
    router = APIRouter(prefix="/audit-log", tags=["audit"])

    @router.get("", response_model=AuditLogPage,
                dependencies=[Depends(require_role("operator", "admin"))])
    async def get_audit_log(
        page: int = Query(1, ge=1),
        per_page: int = Query(50, ge=1, le=200),
        session: AsyncSession = Depends(get_async_session),
    ):
        offset = (page - 1) * per_page
        total_q = await session.execute(select(func.count()).select_from(AuditLog))
        total = total_q.scalar_one()
        rows_q = await session.execute(
            select(AuditLog).order_by(AuditLog.timestamp.desc()).offset(offset).limit(per_page)
        )
        entries = rows_q.scalars().all()
        return AuditLogPage(total=total, page=page, per_page=per_page, entries=entries)
    ```
